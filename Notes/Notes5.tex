\section{Poisson Processes}
% =========================================================

In this section, we study Poisson processes. A Poisson process is a continuous-time counting process used to model random events occurring over time, such as arrivals of customers, phone calls, radioactive emissions, or births. Before defining the Poisson process, we recall two important ingredients: the exponential distribution and the Poisson distribution.

\begin{definition}[Exponential distribution]
	A nonnegative random variable \(X\) is said to have an exponential distribution with rate \(\lambda>0\), written \(X\sim \operatorname{Exp}(\lambda)\), if its density is
	\[
	f_X(x)
	=
	\begin{cases}
		\lambda e^{-\lambda x}, & x\ge 0,\\
		0, & x<0.
	\end{cases}
	\]
\end{definition}

For \(X\sim \operatorname{Exp}(\lambda)\), the distribution function is
\[
F_X(x)=\mathbb{P}(X\le x)
=
\begin{cases}
	1-e^{-\lambda x}, & x\ge 0,\\
	0, & x<0.
\end{cases}
\]
Hence \(\mathbb{P}(X>x)=e^{-\lambda x}\) for \(x\ge 0\). Also,
\[
\mathbb{E}[X]=\frac{1}{\lambda},
\qquad
\operatorname{Var}(X)=\frac{1}{\lambda^2},
\qquad
M_X(t)=\mathbb{E}[e^{tX}]=\frac{\lambda}{\lambda-t},\quad t<\lambda.
\]

\begin{definition}[Memoryless property]
	A nonnegative random variable \(X\) is said to be memoryless if
	\[
	\mathbb{P}(X>t+s\mid X>t)=\mathbb{P}(X>s)
	\]
	for all \(s,t\ge 0\).
\end{definition}

\begin{remark}
	The memoryless property means that the fact that we have already waited for time \(t\) does not influence how much longer we still need to wait. For example, if \(X\) is the waiting time until the next arrival, then \(\mathbb{P}(X>t+s\mid X>t)\) is the probability that we still need to wait more than \(s\) additional units of time, given that we have already waited \(t\) units of time.
\end{remark}

\begin{proposition}
	If \(X\sim \operatorname{Exp}(\lambda)\), then \(X\) is memoryless.
\end{proposition}

\begin{proof}
	For \(s,t\ge 0\),
	\[
	\mathbb{P}(X>t+s\mid X>t)
	=
	\frac{\mathbb{P}(X>t+s)}{\mathbb{P}(X>t)}
	=
	\frac{e^{-\lambda(t+s)}}{e^{-\lambda t}}
	=
	e^{-\lambda s}
	=
	\mathbb{P}(X>s).
	\]
	Hence \(X\) is memoryless.
\end{proof}

\begin{proposition}[Small-time characterization of exponential waiting times]
	Let \(X\) be a nonnegative random variable. Then \(X\sim \operatorname{Exp}(\lambda)\) if and only if, for every \(t\ge 0\),
	\[
	\mathbb{P}(X\le t+h\mid X>t)=\lambda h+o(h),
	\qquad h\downarrow 0.
	\]
\end{proposition}

\begin{remark}
	The notation \(o(h)\) means any function of \(h\) such that
	\[
	\lim_{h\downarrow 0}\frac{o(h)}{h}=0.
	\]
	Thus \(\lambda h+o(h)\) means that, for a very small time interval of length \(h\), the probability that the waiting time ends in that interval is approximately \(\lambda h\).
\end{remark}

\begin{proof}
	First suppose that \(X\sim \operatorname{Exp}(\lambda)\). By the memoryless property,
	\[
	\mathbb{P}(X\le t+h\mid X>t)
	=
	\mathbb{P}(X\le h)
	=
	1-e^{-\lambda h}.
	\]
	Using \(e^{-\lambda h}=1-\lambda h+o(h)\), we get
	\[
	1-e^{-\lambda h}=\lambda h+o(h).
	\]
	Hence \(\mathbb{P}(X\le t+h\mid X>t)=\lambda h+o(h)\).
	
	Conversely, suppose that \(\mathbb{P}(X\le t+h\mid X>t)=\lambda h+o(h)\). Let \(G(t):=\mathbb{P}(X>t)\). Then
	\[
	\mathbb{P}(X>t+h\mid X>t)=1-\lambda h+o(h),
	\]
	so
	\[
	\frac{G(t+h)}{G(t)}=1-\lambda h+o(h).
	\]
	Equivalently,
	\[
	G(t+h)=G(t)\bigl(1-\lambda h+o(h)\bigr).
	\]
	Therefore,
	\[
	\frac{G(t+h)-G(t)}{h}
	=
	-\lambda G(t)+G(t)\frac{o(h)}{h}.
	\]
	Letting \(h\downarrow 0\) gives \(G'(t)=-\lambda G(t)\). Solving this differential equation gives \(G(t)=Ce^{-\lambda t}\). Since \(G(0)=\mathbb{P}(X>0)=1\), we have \(C=1\). Hence
	\[
	\mathbb{P}(X>t)=e^{-\lambda t},
	\]
	so \(X\sim \operatorname{Exp}(\lambda)\).
\end{proof}

\begin{definition}[Poisson distribution]
	A random variable \(X\) has a Poisson distribution with parameter \(\alpha>0\), written \(X\sim \operatorname{Poisson}(\alpha)\), if
	\[
	\mathbb{P}(X=j)=e^{-\alpha}\frac{\alpha^j}{j!},
	\qquad j=0,1,2,\dots.
	\]
\end{definition}

If \(X\sim \operatorname{Poisson}(\alpha)\), then
\[
\mathbb{E}[X]=\alpha,
\qquad
\operatorname{Var}(X)=\alpha.
\]
The moment generating function is
\[
M_X(u)=\mathbb{E}[e^{uX}]=\exp\{\alpha(e^u-1)\}.
\]
Equivalently, the probability generating function is
\[
G_X(s)=\mathbb{E}[s^X]=\exp\{\alpha(s-1)\}.
\]

\begin{proposition}[Sum of independent Poisson random variables]
	Let \(X_1,\dots,X_n\) be independent random variables such that \(X_i\sim \operatorname{Poisson}(\alpha_i)\). Then
	\[
	X_1+\cdots+X_n
	\sim
	\operatorname{Poisson}(\alpha_1+\cdots+\alpha_n).
	\]
\end{proposition}

\begin{proof}
	Using moment generating functions,
	\[
	M_{X_1+\cdots+X_n}(u)
	=
	M_{X_1}(u)\cdots M_{X_n}(u).
	\]
	Since \(M_{X_i}(u)=\exp\{\alpha_i(e^u-1)\}\), we obtain
	\[
	M_{X_1+\cdots+X_n}(u)
	=
	\exp\{(\alpha_1+\cdots+\alpha_n)(e^u-1)\}.
	\]
	This is the moment generating function of a Poisson random variable with parameter \(\alpha_1+\cdots+\alpha_n\). Therefore
	\[
	X_1+\cdots+X_n
	\sim
	\operatorname{Poisson}(\alpha_1+\cdots+\alpha_n).
	\]
\end{proof}

\begin{definition}[Continuous-time discrete-valued stochastic process]
	A stochastic process \(\{X(t):t\ge 0\}\) is called a continuous-time discrete-valued stochastic process if the time parameter \(t\) is continuous, while for each fixed \(t\), the random variable \(X(t)\) takes values in a discrete state space.
\end{definition}

\begin{definition}[Counting process]
	A stochastic process \(\{N(t):t\ge 0\}\) is called a counting process if \(N(t)\) represents the total number of events that have occurred by time \(t\).
\end{definition}

\begin{example}
	The following are examples of counting processes:
	\begin{itemize}[leftmargin=2em]
		\item \(N(t)\) is the number of customers who have entered a store by time \(t\).
		\item \(N(t)\) is the number of babies born in a hospital by time \(t\).
		\item \(N(t)\) is the number of phone calls received by time \(t\).
		\item \(N(t)\) is the number of radioactive particles emitted by time \(t\).
	\end{itemize}
\end{example}

A general counting process satisfies the following properties:
\begin{itemize}[leftmargin=2em]
	\item \(N(t)\ge 0\) for all \(t\ge 0\).
	\item \(N(t)\) is integer-valued.
	\item If \(s<t\), then \(N(s)\le N(t)\).
	\item If \(s<t\), then \(N(t)-N(s)\) represents the number of events occurring during the time interval \((s,t]\).
\end{itemize}

\begin{remark}
	In a general counting process, the increments \(N(t_1)-N(s_1)\) and \(N(t_2)-N(s_2)\) need not be independent, even if the intervals \((s_1,t_1]\) and \((s_2,t_2]\) are disjoint. Also, the distribution of \(N(t)-N(s)\) may depend on the actual values of \(s\) and \(t\), not just on the length \(t-s\).
\end{remark}

\begin{definition}[Independent increments]
	A counting process \(\{N(t):t\ge 0\}\) is said to have independent increments if the numbers of events occurring in disjoint time intervals are independent. That is, if
	\[
	0\le t_0<t_1<\cdots<t_n,
	\]
	then the random variables
	\[
	N(t_1)-N(t_0),\quad
	N(t_2)-N(t_1),\quad
	\dots,\quad
	N(t_n)-N(t_{n-1})
	\]
	are independent.
\end{definition}

\begin{definition}[Stationary increments]
	A counting process \(\{N(t):t\ge 0\}\) is said to have stationary increments if the distribution of \(N(t)-N(s)\) depends only on the length of the interval \(t-s\). Equivalently, for all \(s,t\ge 0\), \(N(s+t)-N(s)\) has the same distribution as \(N(t)-N(0)\).
\end{definition}

\begin{definition}[Poisson process, first definition]
	A counting process \(\{N(t):t\ge 0\}\) is called a Poisson process with rate \(\lambda>0\) if:
	\begin{enumerate}[leftmargin=2em]
		\item \(N(0)=0\).
		\item The process has independent increments.
		\item For all \(s,t\ge 0\),
		\[
		N(s+t)-N(s)
		\sim
		\operatorname{Poisson}(\lambda t).
		\]
	\end{enumerate}
\end{definition}

\begin{remark}
	The third condition says two things at once: the number of events in an interval of length \(t\) has Poisson distribution with parameter \(\lambda t\), and the distribution depends only on the length of the interval, not on its location. Thus the process has stationary increments.
\end{remark}

\begin{definition}[Poisson process, second definition]
	A counting process \(\{N(t):t\ge 0\}\) is called a Poisson process with rate \(\lambda>0\) if:
	\begin{enumerate}[leftmargin=2em]
		\item \(N(0)=0\).
		\item The process has stationary and independent increments.
		\item As \(h\downarrow 0\),
		\[
		\mathbb{P}(N(h)=1)=\lambda h+o(h).
		\]
		\item As \(h\downarrow 0\),
		\[
		\mathbb{P}(N(h)\ge 2)=o(h).
		\]
	\end{enumerate}
\end{definition}

\begin{remark}
	The condition \(\mathbb{P}(N(h)=1)=\lambda h+o(h)\) means that, in a very small time interval of length \(h\), the probability of one event is approximately \(\lambda h\). The condition \(\mathbb{P}(N(h)\ge 2)=o(h)\) means that the probability of two or more events in a very small interval is negligible compared with \(h\).
\end{remark}

\begin{remark}
	From the two small-time conditions, we also get
	\[
	\mathbb{P}(N(h)=0)=1-\lambda h+o(h).
	\]
	Indeed,
	\[
	\mathbb{P}(N(h)=0)
	=
	1-\mathbb{P}(N(h)=1)-\mathbb{P}(N(h)\ge 2)
	=
	1-\lambda h+o(h).
	\]
\end{remark}

\begin{theorem}
	The two definitions of a Poisson process are equivalent.
\end{theorem}

\begin{proof}
	We prove both directions.
	
	First assume that \(\{N(t):t\ge 0\}\) is a Poisson process according to the first definition. Then \(N(0)=0\) and the process has independent increments. Also,
	\[
	N(s+t)-N(s)
	\sim
	\operatorname{Poisson}(\lambda t),
	\]
	so the increments are stationary.
	
	It remains to verify the two small-time conditions. Since \(N(h)\sim \operatorname{Poisson}(\lambda h)\), we have
	\[
	\mathbb{P}(N(h)=1)=e^{-\lambda h}\lambda h.
	\]
	Using \(e^{-\lambda h}=1-\lambda h+o(h)\), we get
	\[
	\mathbb{P}(N(h)=1)
	=
	\lambda h(1-\lambda h+o(h))
	=
	\lambda h+o(h).
	\]
	Also,
	\[
	\mathbb{P}(N(h)\ge 2)
	=
	1-\mathbb{P}(N(h)=0)-\mathbb{P}(N(h)=1).
	\]
	Since \(\mathbb{P}(N(h)=0)=e^{-\lambda h}=1-\lambda h+o(h)\) and \(\mathbb{P}(N(h)=1)=\lambda h+o(h)\), we obtain
	\[
	\mathbb{P}(N(h)\ge 2)
	=
	1-\bigl(1-\lambda h+o(h)\bigr)-\bigl(\lambda h+o(h)\bigr)
	=
	o(h).
	\]
	Thus the second definition holds.
	
	Conversely, assume that \(\{N(t):t\ge 0\}\) satisfies the second definition. We show that \(N(t)\sim \operatorname{Poisson}(\lambda t)\) for every \(t\ge 0\). For \(0\le s\le 1\), define the probability generating function
	\[
	G(t,s):=\mathbb{E}[s^{N(t)}].
	\]
	Using stationary and independent increments,
	\[
	N(t+h)=N(t)+\bigl(N(t+h)-N(t)\bigr),
	\]
	where \(N(t)\) is independent of \(N(t+h)-N(t)\), and \(N(t+h)-N(t)\) has the same distribution as \(N(h)\). Hence
	\[
	G(t+h,s)=G(t,s)G(h,s).
	\]
	
	Now compute \(G(h,s)\) for small \(h\). By the small-time conditions,
	\[
	\mathbb{P}(N(h)=0)=1-\lambda h+o(h),
	\qquad
	\mathbb{P}(N(h)=1)=\lambda h+o(h),
	\qquad
	\mathbb{P}(N(h)\ge 2)=o(h).
	\]
	Therefore, for \(0\le s\le 1\),
	\[
	G(h,s)
	=
	(1-\lambda h+o(h))
	+
	s(\lambda h+o(h))
	+
	o(h)
	=
	1+\lambda h(s-1)+o(h).
	\]
	Substituting this into \(G(t+h,s)=G(t,s)G(h,s)\), we get
	\[
	G(t+h,s)
	=
	G(t,s)\bigl(1+\lambda h(s-1)+o(h)\bigr).
	\]
	Hence
	\[
	\frac{G(t+h,s)-G(t,s)}{h}
	=
	\lambda(s-1)G(t,s)+G(t,s)\frac{o(h)}{h}.
	\]
	Letting \(h\downarrow 0\) gives the differential equation
	\[
	\frac{\partial}{\partial t}G(t,s)=\lambda(s-1)G(t,s).
	\]
	Since \(N(0)=0\), we have \(G(0,s)=\mathbb{E}[s^{N(0)}]=1\). Solving the differential equation gives
	\[
	G(t,s)=\exp\{\lambda t(s-1)\}.
	\]
	This is exactly the probability generating function of a Poisson random variable with parameter \(\lambda t\). Therefore \(N(t)\sim \operatorname{Poisson}(\lambda t)\).
	
	By stationary increments, the same argument gives
	\[
	N(s+t)-N(s)
	\sim
	\operatorname{Poisson}(\lambda t)
	\]
	for all \(s,t\ge 0\). Hence the first definition holds.
\end{proof}

\begin{remark}
	The parameter \(\lambda\) is called the rate or intensity of the Poisson process. It represents the average number of events per unit time. For example, if customers arrive according to a Poisson process with rate \(\lambda=5\) per hour, then the expected number of arrivals in one hour is \(5\), the expected number of arrivals in \(t\) hours is \(5t\), and the number of arrivals in an interval of length \(t\) has distribution \(\operatorname{Poisson}(5t)\).
\end{remark}

\begin{proposition}
	If \(\{N(t):t\ge 0\}\) is a Poisson process with rate \(\lambda\), then
	\[
	\mathbb{E}[N(t)]=\lambda t,
	\qquad
	\operatorname{Var}(N(t))=\lambda t.
	\]
\end{proposition}

\begin{proof}
	Since \(N(t)\sim \operatorname{Poisson}(\lambda t)\), the mean and variance of \(N(t)\) are both equal to the Poisson parameter \(\lambda t\). Therefore
	\[
	\mathbb{E}[N(t)]=\lambda t,
	\qquad
	\operatorname{Var}(N(t))=\lambda t.
	\]
\end{proof}

	\section{Interarrival Times, Waiting Times, and Exponential Race}
	% =========================================================
	
	We continue our study of Poisson processes by looking at the times between arrivals and some useful consequences of the exponential distribution. Let \(\{N(t):t\ge 0\}\) be a Poisson process with rate \(\lambda>0\).
	
	\begin{definition}[Interarrival times]
		For \(n\ge 1\), let \(T_n\) denote the elapsed time between the \((n-1)\)-st event and the \(n\)-th event. The sequence \(T_1,T_2,\dots\) is called the sequence of interarrival times.
	\end{definition}
	
	\begin{proposition}
		The interarrival times \(T_1,T_2,\dots\) are independent and identically distributed exponential random variables with rate \(\lambda\). That is,
		\[
		T_1,T_2,\dots \text{ are i.i.d. } \operatorname{Exp}(\lambda).
		\]
	\end{proposition}
	
	\begin{proof}
		The event \(\{T_1>t\}\) means that no event occurs during \((0,t]\). Hence
		\[
		\mathbb{P}(T_1>t)=\mathbb{P}(N(t)=0)=e^{-\lambda t},
		\]
		so \(T_1\sim\operatorname{Exp}(\lambda)\).
		
		Now suppose the first event occurs at time \(s\). By stationary and independent increments, the process after time \(s\) behaves like a fresh Poisson process and is independent of the past. Thus, for \(t\ge 0\),
		\[
		\mathbb{P}(T_2>t\mid T_1=s)
		=
		\mathbb{P}(N(s+t)-N(s)=0)
		=
		e^{-\lambda t}.
		\]
		Hence \(T_2\sim\operatorname{Exp}(\lambda)\), independently of \(T_1\). Repeating the same argument after each arrival gives the result.
	\end{proof}
	
	\begin{remark}
		This gives another way to construct a Poisson process. Take i.i.d. random variables \(T_1,T_2,\dots\sim\operatorname{Exp}(\lambda)\), define
		\[
		S_n=T_1+\cdots+T_n,
		\]
		and then define
		\[
		N(t)=\max\{n:S_n\le t\}.
		\]
		Then \(\{N(t):t\ge 0\}\) is a Poisson process with rate \(\lambda\).
	\end{remark}
	
	\begin{definition}[Waiting time]
		The arrival time of the \(n\)-th event is denoted by \(S_n\) and is called the waiting time until the \(n\)-th event. Thus
		\[
		S_n=T_1+\cdots+T_n.
		\]
	\end{definition}
	
	\begin{proposition}
		For \(n\ge 1\), the waiting time \(S_n\) has Gamma distribution with shape parameter \(n\) and rate parameter \(\lambda\). That is,
		\[
		S_n\sim \operatorname{Gamma}(n,\lambda),
		\]
		with density
		\[
		f_{S_n}(t)=\frac{\lambda^n t^{n-1}e^{-\lambda t}}{(n-1)!},
		\qquad t\ge 0.
		\]
	\end{proposition}
	
	\begin{proof}
		Since \(S_n=T_1+\cdots+T_n\), where \(T_1,\dots,T_n\) are i.i.d. \(\operatorname{Exp}(\lambda)\), the sum has Gamma distribution with the stated density.
	\end{proof}
	
	\begin{remark}
		There is a useful relation between \(S_n\) and \(N(t)\):
		\[
		\{S_n\le t\}=\{N(t)\ge n\}.
		\]
		Therefore,
		\[
		\mathbb{P}(S_n\le t)
		=
		\mathbb{P}(N(t)\ge n)
		=
		1-\sum_{k=0}^{n-1}e^{-\lambda t}\frac{(\lambda t)^k}{k!}.
		\]
	\end{remark}
	
	\begin{remark}[Direct derivation of the density of \(S_n\)]
		For small \(h>0\), the event \(t<S_n\le t+h\) mainly occurs when exactly \(n-1\) events have occurred by time \(t\), and exactly one further event occurs in \((t,t+h]\). More precisely,
		\[
		\mathbb{P}(t<S_n\le t+h)
		=
		\mathbb{P}(N(t)=n-1,\ N(t+h)-N(t)=1)+o(h).
		\]
		Using independent increments,
		\[
		\mathbb{P}(t<S_n\le t+h)
		=
		e^{-\lambda t}\frac{(\lambda t)^{n-1}}{(n-1)!}(\lambda h+o(h))+o(h).
		\]
		Dividing by \(h\) and letting \(h\downarrow 0\) gives
		\[
		f_{S_n}(t)=\frac{\lambda^n t^{n-1}e^{-\lambda t}}{(n-1)!}.
		\]
	\end{remark}
	
	\begin{proposition}[Thinning of a Poisson process]
		Let \(\{N(t):t\ge 0\}\) be a Poisson process with rate \(\lambda\). Suppose each event is independently classified as type I with probability \(p\), and type II with probability \(1-p\). Let \(N_1(t)\) and \(N_2(t)\) denote the numbers of type I and type II events in \((0,t]\), respectively. Then \(N_1\) and \(N_2\) are independent Poisson processes with rates \(\lambda p\) and \(\lambda(1-p)\).
	\end{proposition}
	
	\begin{proof}
		Clearly \(N_1(0)=N_2(0)=0\). Since the original process has stationary and independent increments, and the classifications are made independently for each event, both \(N_1\) and \(N_2\) also have stationary and independent increments.
		
		For small \(h>0\),
		\[
		\mathbb{P}(N_1(h)=1)
		=
		p\mathbb{P}(N(h)=1)+o(h)
		=
		p(\lambda h+o(h))+o(h)
		=
		\lambda p h+o(h).
		\]
		Also, \(\mathbb{P}(N_1(h)\ge 2)\le \mathbb{P}(N(h)\ge 2)=o(h)\). Hence \(N_1\) is a Poisson process with rate \(\lambda p\). Similarly, \(N_2\) is a Poisson process with rate \(\lambda(1-p)\).
		
		To prove independence, for \(m,n\ge 0\), condition on \(N(t)=m+n\):
		\[
		\mathbb{P}(N_1(t)=m,N_2(t)=n)
		=
		\mathbb{P}(N(t)=m+n)\binom{m+n}{m}p^m(1-p)^n.
		\]
		Thus
		\[
		\mathbb{P}(N_1(t)=m,N_2(t)=n)
		=
		e^{-\lambda pt}\frac{(\lambda pt)^m}{m!}
		e^{-\lambda(1-p)t}\frac{(\lambda(1-p)t)^n}{n!},
		\]
		which factors into \(\mathbb{P}(N_1(t)=m)\mathbb{P}(N_2(t)=n)\). Applying the same argument to disjoint intervals gives independence of the two processes.
	\end{proof}
	
	\begin{example}
		Suppose immigrants arrive in an area according to a Poisson process with rate \(10\) per week. If each immigrant is of English descent with probability \(p\), independently of all others, then the number of English immigrants arriving in \(4\) weeks has distribution \(\operatorname{Poisson}(40p)\). Hence the probability that no English immigrants arrive during these \(4\) weeks is \(e^{-40p}\).
	\end{example}
	
	\begin{proposition}[Superposition of independent Poisson processes]
		Let \(N_1(t),\dots,N_r(t)\) be independent Poisson processes with rates \(\lambda_1,\dots,\lambda_r\). Then
		\[
		N(t):=N_1(t)+\cdots+N_r(t)
		\]
		is a Poisson process with rate \(\lambda_1+\cdots+\lambda_r\).
	\end{proposition}
	
	\begin{proof}
		The process \(N(t)\) is a counting process with stationary and independent increments. Moreover,
		\[
		N(s+t)-N(s)
		=
		\sum_{i=1}^{r}\bigl(N_i(s+t)-N_i(s)\bigr).
		\]
		Each increment \(N_i(s+t)-N_i(s)\) has distribution \(\operatorname{Poisson}(\lambda_i t)\), and these increments are independent. Hence their sum is Poisson with parameter \((\lambda_1+\cdots+\lambda_r)t\).
	\end{proof}
	
\begin{proposition}[Exponential race]
	Let \(X_1,\dots,X_n\) be independent random variables with \(X_i\sim\operatorname{Exp}(\lambda_i)\), where \(\lambda_i>0\). Define
	\[
	X=\min\{X_1,\dots,X_n\}.
	\]
	Then
	\[
	X\sim\operatorname{Exp}(\lambda_1+\cdots+\lambda_n),
	\]
	and
	\[
	\mathbb{P}(X=X_i)=\frac{\lambda_i}{\lambda_1+\cdots+\lambda_n},
	\qquad i=1,\dots,n.
	\]
	Moreover, the minimum time \(X\) and the identity of the minimizing variable are independent.
\end{proposition}

\begin{proof}
	For \(t\ge 0\),
	\[
	\mathbb{P}(X>t)
	=
	\mathbb{P}(X_1>t,\dots,X_n>t).
	\]
	Since \(X_1,\dots,X_n\) are independent,
	\[
	\mathbb{P}(X>t)
	=
	\prod_{i=1}^{n}\mathbb{P}(X_i>t)
	=
	\prod_{i=1}^{n}e^{-\lambda_i t}
	=
	e^{-(\lambda_1+\cdots+\lambda_n)t}.
	\]
	Thus
	\[
	X\sim\operatorname{Exp}(\lambda_1+\cdots+\lambda_n).
	\]
	
	Next, we compute the probability that clock \(i\) is the first clock to ring. Since exponential random variables are continuous, ties have probability \(0\). Hence
	\[
	\{X=X_i\}
	=
	\{X_i<X_j\text{ for all }j\neq i\}.
	\]
	We condition on the value of \(X_i\). If \(X_i=t\), then clock \(i\) rings at time \(t\). For clock \(i\) to be the first clock, all other clocks must still be silent at time \(t\), meaning
	\[
	X_j>t
	\qquad
	\text{for all }j\neq i.
	\]
	Since \(X_i\) has density \(\lambda_i e^{-\lambda_i t}\), we get
	\[
	\mathbb{P}(X=X_i)
	=
	\int_0^\infty
	\mathbb{P}(X_j>t\text{ for all }j\neq i)\lambda_i e^{-\lambda_i t}\,dt.
	\]
	By independence,
	\[
	\mathbb{P}(X_j>t\text{ for all }j\neq i)
	=
	\prod_{j\neq i}e^{-\lambda_j t}.
	\]
	Therefore,
	\[
	\mathbb{P}(X=X_i)
	=
	\int_0^\infty
	\left(\prod_{j\neq i}e^{-\lambda_j t}\right)
	\lambda_i e^{-\lambda_i t}\,dt.
	\]
	Combining the exponential terms gives
	\[
	\mathbb{P}(X=X_i)
	=
	\int_0^\infty
	\lambda_i e^{-(\lambda_1+\cdots+\lambda_n)t}\,dt
	=
	\frac{\lambda_i}{\lambda_1+\cdots+\lambda_n}.
	\]
	
	Finally, let \(I\) denote the index of the first clock. Thus \(I=i\) means that clock \(i\) is the first clock to ring. We show that \(X\) and \(I\) are independent.
	
	Consider the event
	\[
	\{X>t,\ I=i\}.
	\]
	This event means that the first ringing time is after time \(t\), and that clock \(i\) is the first clock to ring. Equivalently, clock \(i\) rings at some time \(u>t\), and every other clock has not rung before time \(u\). Thus we integrate over all possible winning times \(u>t\).
	
	If \(X_i=u\), then clock \(i\) rings at time \(u\). For clock \(i\) to be the first clock, we need
	\[
	X_j>u
	\qquad
	\text{for all }j\neq i.
	\]
	The density of \(X_i\) at \(u\) is
	\[
	f_{X_i}(u)=\lambda_i e^{-\lambda_i u}.
	\]
	Therefore,
	\[
	\mathbb{P}(X>t,\ I=i)
	=
	\int_t^\infty
	\mathbb{P}(X_j>u\text{ for all }j\neq i)
	\lambda_i e^{-\lambda_i u}\,du.
	\]
	By independence again,
	\[
	\mathbb{P}(X_j>u\text{ for all }j\neq i)
	=
	\prod_{j\neq i}e^{-\lambda_j u}.
	\]
	Hence
	\[
	\mathbb{P}(X>t,\ I=i)
	=
	\int_t^\infty
	\left(\prod_{j\neq i}e^{-\lambda_j u}\right)
	\lambda_i e^{-\lambda_i u}\,du.
	\]
	Combining the exponential terms,
	\[
	\mathbb{P}(X>t,\ I=i)
	=
	\int_t^\infty
	\lambda_i e^{-(\lambda_1+\cdots+\lambda_n)u}\,du.
	\]
	Evaluating the integral gives
	\[
	\mathbb{P}(X>t,\ I=i)
	=
	\frac{\lambda_i}{\lambda_1+\cdots+\lambda_n}
	e^{-(\lambda_1+\cdots+\lambda_n)t}.
	\]
	But
	\[
	\mathbb{P}(I=i)
	=
	\frac{\lambda_i}{\lambda_1+\cdots+\lambda_n},
	\]
	and
	\[
	\mathbb{P}(X>t)
	=
	e^{-(\lambda_1+\cdots+\lambda_n)t}.
	\]
	Therefore,
	\[
	\mathbb{P}(X>t,\ I=i)
	=
	\mathbb{P}(I=i)\mathbb{P}(X>t).
	\]
	This shows that \(X\) and \(I\) are independent.
\end{proof}

	
	\begin{remark}
		This result is often called an exponential race. If several independent exponential clocks are running, then the first ringing time is exponential with rate equal to the sum of the rates, and clock \(i\) rings first with probability proportional to \(\lambda_i\).
	\end{remark}
	
	\begin{corollary}
		If \(X\sim\operatorname{Exp}(\lambda)\) and \(Y\sim\operatorname{Exp}(\mu)\) are independent, then
		\[
		\mathbb{P}(X<Y)=\frac{\lambda}{\lambda+\mu},
		\qquad
		\mathbb{P}(Y<X)=\frac{\mu}{\lambda+\mu}.
		\]
	\end{corollary}
	
	\begin{example}[Competing Poisson processes]
		Let \(N_1(t)\) and \(N_2(t)\) be independent Poisson processes with rates \(\lambda_1\) and \(\lambda_2\). Let \(S_n^{(1)}\) be the time of the \(n\)-th event of process \(1\), and let \(S_m^{(2)}\) be the time of the \(m\)-th event of process \(2\). We compute \(\mathbb{P}(S_n^{(1)}<S_m^{(2)})\).
		
		By superposition, \(N(t)=N_1(t)+N_2(t)\) is a Poisson process with rate \(\lambda_1+\lambda_2\). Each event of the combined process belongs to process \(1\) with probability
		\[
		p=\frac{\lambda_1}{\lambda_1+\lambda_2},
		\]
		and to process \(2\) with probability \(1-p=\lambda_2/(\lambda_1+\lambda_2)\), independently of the other events.
		
		The event \(S_n^{(1)}<S_m^{(2)}\) means that the \(n\)-th type \(1\) event occurs before the \(m\)-th type \(2\) event. Equivalently, among the first \(n+m-1\) events of the combined process, there must be at least \(n\) type \(1\) events. Therefore,
		\[
		\mathbb{P}(S_n^{(1)}<S_m^{(2)})
		=
		\sum_{k=n}^{n+m-1}
		\binom{n+m-1}{k}p^k(1-p)^{n+m-1-k}.
		\]
	\end{example}
	
	\begin{remark}
		The previous formula is the same as a coin-tossing problem. Each event of the combined Poisson process is labeled type \(1\) with probability \(p\) and type \(2\) with probability \(1-p\). We ask for the probability that \(n\) type \(1\) events occur before \(m\) type \(2\) events occur.
	\end{remark}
	
	\begin{example}
		In the special case \(n=m=1\),
		\[
		\mathbb{P}(S_1^{(1)}<S_1^{(2)})
		=
		p
		=
		\frac{\lambda_1}{\lambda_1+\lambda_2}.
		\]
		In the case \(n=2\) and \(m=1\), the first two events of the combined process must both come from process \(1\). Thus
		\[
		\mathbb{P}(S_2^{(1)}<S_1^{(2)})
		=
		p^2
		=
		\left(\frac{\lambda_1}{\lambda_1+\lambda_2}\right)^2.
		\]
	\end{example}
	
	\begin{example}[Coupon collector problem]
		There are \(m\) different types of coupons. Each time a coupon is collected, it is of type \(j\) with probability \(p_j\), independently of previous coupons, where \(p_1+\cdots+p_m=1\). Let \(N\) be the number of coupons needed to obtain a complete collection. We compute \(\mathbb{E}[N]\).
		
		Suppose coupons arrive according to a Poisson process with rate \(1\). By thinning, type \(j\) coupons form an independent Poisson process with rate \(p_j\). Let \(X_j\) be the time of the first type \(j\) coupon. Then
		\[
		X_j\sim\operatorname{Exp}(p_j),
		\qquad
		X_1,\dots,X_m \text{ are independent}.
		\]
		The time at which the collection is complete is \(X=\max\{X_1,\dots,X_m\}\). Hence
		\[
		\mathbb{P}(X\le t)
		=
		\prod_{j=1}^{m}(1-e^{-p_jt}).
		\]
		
		Using the tail expectation formula for nonnegative random variables,
		\[
		\mathbb{E}[X]
		=
		\int_0^\infty \mathbb{P}(X>t)\,dt
		=
		\int_0^\infty
		\left[
		1-\prod_{j=1}^{m}(1-e^{-p_jt})
		\right]dt.
		\]
		Since the overall coupon arrival process has rate \(1\), the expected completion time equals the expected number of coupons needed. Thus
		\[
		\mathbb{E}[N]
		=
		\int_0^\infty
		\left[
		1-\prod_{j=1}^{m}(1-e^{-p_jt})
		\right]dt.
		\]
	\end{example}
	
	\begin{remark}
		If the overall coupon arrival process has rate \(\mu\) instead of rate \(1\), then type \(j\) coupons form an independent Poisson process with rate \(\mu p_j\). The completion time is sped up by a factor of \(\mu\), but the sequence of coupon types is unchanged. Hence the expected number of coupons needed is independent of \(\mu\).
	\end{remark}
	
	\begin{corollary}[Coupon collector with equally likely types]
		If \(p_j=1/m\) for all \(j=1,\dots,m\), then
		\[
		\mathbb{E}[N]
		=
		m\left(1+\frac12+\cdots+\frac1m\right)
		=
		mH_m,
		\]
		where \(H_m\) is the \(m\)-th harmonic number.
	\end{corollary}
	
	\begin{proof}
		In the equally likely case,
		\[
		\mathbb{E}[N]
		=
		\int_0^\infty \left[1-(1-e^{-t/m})^m\right]dt.
		\]
		Using
		\[
		(1-e^{-t/m})^m
		=
		\sum_{k=0}^{m}\binom{m}{k}(-1)^k e^{-kt/m},
		\]
		we get
		\[
		\mathbb{E}[N]
		=
		\sum_{k=1}^{m}(-1)^{k+1}\binom{m}{k}\int_0^\infty e^{-kt/m}\,dt.
		\]
		Therefore,
		\[
		\mathbb{E}[N]
		=
		m\sum_{k=1}^{m}(-1)^{k+1}\binom{m}{k}\frac1k
		=
		mH_m.
		\]
	\end{proof}
	
	\begin{example}[Number of coupon types appearing exactly once]
		In the coupon collector problem with \(m\) equally likely types, let \(Y\) be the number of types that appear exactly once in the completed collection. We compute \(\mathbb{E}[Y]\).
		
		Let \(I_i\) be the indicator of the event that type \(i\) appears exactly once in the completed collection. Then \(Y=I_1+\cdots+I_m\), so by symmetry,
		\[
		\mathbb{E}[Y]=m\mathbb{E}[I_1]=m\mathbb{P}(I_1=1).
		\]
		
		In the Poissonized model, type \(i\) coupons form a Poisson process with rate \(1/m\). Let \(S_i\) be the time of the second type \(i\) coupon. Then \(S_i\sim\operatorname{Gamma}(2,1/m)\), with density
		\[
		f_{S_i}(s)=\frac{s}{m^2}e^{-s/m},
		\qquad s\ge 0.
		\]
		For each \(j\neq i\), let \(X_j\) be the first arrival time of a type \(j\) coupon. Then \(X_j\sim\operatorname{Exp}(1/m)\), independently over \(j\neq i\).
		
		The event \(I_i=1\) occurs exactly when every other type appears before the second type \(i\) coupon arrives:
		\[
		I_i=1
		\quad\Longleftrightarrow\quad
		X_j<S_i \text{ for every } j\neq i.
		\]
		Conditioning on \(S_i=s\),
		\[
		\mathbb{P}(I_i=1\mid S_i=s)
		=
		(1-e^{-s/m})^{m-1}.
		\]
		Therefore,
		\[
		\mathbb{P}(I_i=1)
		=
		\int_0^\infty
		(1-e^{-s/m})^{m-1}\frac{s}{m^2}e^{-s/m}\,ds.
		\]
		Using the change of variables \(u=e^{-s/m}\), we obtain
		\[
		\mathbb{P}(I_i=1)
		=
		-\int_0^1 (1-u)^{m-1}\log u\,du.
		\]
		Using the identity
		\[
		-\int_0^1 (1-u)^{m-1}\log u\,du
		=
		\frac{H_m}{m},
		\]
		we get \(\mathbb{P}(I_i=1)=H_m/m\). Hence
		\[
		\mathbb{E}[Y]
		=
		m\cdot \frac{H_m}{m}
		=
		H_m.
		\]
	\end{example}

	\section{Conditional Distribution of the Waiting/Arrival Times}
	% =========================================================
	
	Let \(\{N(t):t\ge 0\}\) be a Poisson process with rate \(\lambda>0\). Let \(T_1,T_2,\dots\) be the interarrival times and let
	\[
	S_n=T_1+\cdots+T_n
	\]
	be the arrival time of the \(n\)-th event.
	
	We first consider a simple question: suppose we know that exactly one event occurs in \((0,t]\). What is the conditional distribution of the arrival time \(S_1\)?
	
	\begin{proposition}
		Given that \(N(t)=1\), the first arrival time \(S_1\) is uniformly distributed on \([0,t]\). That is,
		\[
		(S_1\mid N(t)=1)\sim U(0,t).
		\]
	\end{proposition}
	
	\begin{proof}
		For \(0\le s\le t\),
		\[
		\mathbb{P}(S_1\le s\mid N(t)=1)
		=
		\frac{\mathbb{P}(S_1\le s,\ N(t)=1)}{\mathbb{P}(N(t)=1)}.
		\]
		The event \(\{S_1\le s,\ N(t)=1\}\) means that there is exactly one event in \((0,s]\) and no event in \((s,t]\). Hence, by independent increments,
		\begin{align*}
			\mathbb{P}(S_1\le s,\ N(t)=1)
			&=\mathbb{P}(N(s)=1,\ N(t)-N(s)=0)\\
			&=\mathbb{P}(N(s)=1)\mathbb{P}(N(t)-N(s)=0)\\
			&=e^{-\lambda s}\lambda s\cdot e^{-\lambda(t-s)}
			=\lambda s e^{-\lambda t}.
		\end{align*}
		Also,
		\[
		\mathbb{P}(N(t)=1)=\lambda t e^{-\lambda t}.
		\]
		Therefore,
		\[
		\mathbb{P}(S_1\le s\mid N(t)=1)=\frac{\lambda s e^{-\lambda t}}{\lambda t e^{-\lambda t}}=\frac{s}{t}.
		\]
		This is the c.d.f. of \(U(0,t)\), so \((S_1\mid N(t)=1)\sim U(0,t)\).
	\end{proof}
	
	\begin{remark}
		The conclusion is intuitive: if we only know that exactly one event occurred during \((0,t]\), then the event is equally likely to have occurred anywhere in the interval.
	\end{remark}
	
	\begin{definition}[Order statistics]
		Let \(Y_1,\dots,Y_n\) be random variables. Their order statistics are denoted by
		\[
		Y_{(1)}\le Y_{(2)}\le \cdots\le Y_{(n)},
		\]
		where \(Y_{(k)}\) is the \(k\)-th smallest value among \(Y_1,\dots,Y_n\).
	\end{definition}
	
	\begin{proposition}[Joint density of order statistics]
		Let \(Y_1,\dots,Y_n\) be i.i.d. continuous random variables with density \(f\). Then the joint density of the order statistics \(Y_{(1)},\dots,Y_{(n)}\) is
		\[
		f_{Y_{(1)},\dots,Y_{(n)}}(y_1,\dots,y_n)
		=
		n!\prod_{i=1}^{n}f(y_i),
		\qquad y_1<y_2<\cdots<y_n.
		\]
		The density is \(0\) otherwise.
	\end{proposition}
	
	\begin{proof}
		For fixed \(y_1<\cdots<y_n\), the unordered vector \((Y_1,\dots,Y_n)\) can be arranged as \((y_1,\dots,y_n)\) in \(n!\) different ways. Since the \(Y_i\)'s are independent, each ordering has density \(\prod_{i=1}^{n}f(y_i)\). Therefore the joint density of the ordered vector is
		\[
		n!\prod_{i=1}^{n}f(y_i),
		\qquad y_1<\cdots<y_n.
		\]
	\end{proof}
	
	\begin{example}[Uniform order statistics]
		If \(Y_1,\dots,Y_n\) are i.i.d. \(U(0,t)\), then \(f(y)=1/t\) for \(0<y<t\). Hence the joint density of \(Y_{(1)},\dots,Y_{(n)}\) is
		\[
		f_{Y_{(1)},\dots,Y_{(n)}}(y_1,\dots,y_n)
		=
		\frac{n!}{t^n},
		\qquad 0<y_1<\cdots<y_n<t.
		\]
	\end{example}
	
\begin{proposition}
	Given that \(S_n=t\), the previous arrival times \(S_1,\dots,S_{n-1}\) have the same distribution as the order statistics of \(n-1\) independent \(U(0,t)\) random variables.
\end{proposition}

\begin{proof}
	Recall that \(S_k=T_1+\cdots+T_k\), where \(T_1,T_2,\dots\) are i.i.d. \(\operatorname{Exp}(\lambda)\). Hence the joint density of \(T_1,\dots,T_n\) is
	\[
	f_{T_1,\dots,T_n}(t_1,\dots,t_n)
	=
	\lambda^n e^{-\lambda(t_1+\cdots+t_n)},
	\qquad t_1,\dots,t_n>0.
	\]
	We now derive the joint density of \(S_1,\dots,S_n\). Since
	\[
	S_1=T_1,\quad S_2=T_1+T_2,\quad \dots,\quad S_n=T_1+\cdots+T_n,
	\]
	the inverse transformation is
	\[
	T_1=S_1,\quad T_2=S_2-S_1,\quad \dots,\quad T_n=S_n-S_{n-1}.
	\]
	Thus \(T_i>0\) for all \(i\) is equivalent to \(0<S_1<S_2<\cdots<S_n\). Also,
	\[
	T_1+\cdots+T_n
	=
	S_1+(S_2-S_1)+\cdots+(S_n-S_{n-1})
	=
	S_n.
	\]
	The Jacobian matrix of the inverse transformation from \((s_1,\dots,s_n)\) to \((t_1,\dots,t_n)\) is
	\[
	\frac{\partial(t_1,\dots,t_n)}{\partial(s_1,\dots,s_n)}
	=
	\begin{pmatrix}
		1 & 0 & 0 & \cdots & 0 \\
		-1 & 1 & 0 & \cdots & 0 \\
		0 & -1 & 1 & \cdots & 0 \\
		\vdots & \vdots & \vdots & \ddots & \vdots \\
		0 & 0 & 0 & \cdots & 1
	\end{pmatrix}.
	\]
	This matrix is lower triangular with diagonal entries all equal to \(1\), so its determinant is \(1\). Therefore, by the change-of-variables formula,
	\[
	f_{S_1,\dots,S_n}(s_1,\dots,s_n)
	=
	f_{T_1,\dots,T_n}(s_1,s_2-s_1,\dots,s_n-s_{n-1})
	\cdot 1.
	\]
	Using \(t_1+\cdots+t_n=s_n\), we get
	\[
	f_{S_1,\dots,S_n}(s_1,\dots,s_n)
	=
	\lambda^n e^{-\lambda s_n},
	\qquad 0<s_1<\cdots<s_n.
	\]
	In particular, for \(0<s_1<\cdots<s_{n-1}<t\),
	\[
	f_{S_1,\dots,S_n}(s_1,\dots,s_{n-1},t)
	=
	\lambda^n e^{-\lambda t}.
	\]
	
	Next, the density of \(S_n\) is the Gamma density
	\[
	f_{S_n}(t)
	=
	\frac{\lambda^n t^{n-1}e^{-\lambda t}}{(n-1)!},
	\qquad t>0.
	\]
	
	Therefore the conditional joint density of \(S_1,\dots,S_{n-1}\) given \(S_n=t\) is
	\[
	f_{S_1,\dots,S_{n-1}\mid S_n=t}(s_1,\dots,s_{n-1})
	=
	\frac{
		f_{S_1,\dots,S_n}(s_1,\dots,s_{n-1},t)
	}{
		f_{S_n}(t)
	}.
	\]
	Substituting the two densities gives
	\[
	f_{S_1,\dots,S_{n-1}\mid S_n=t}(s_1,\dots,s_{n-1})
	=
	\frac{\lambda^n e^{-\lambda t}}
	{\lambda^n t^{n-1}e^{-\lambda t}/(n-1)!}
	=
	\frac{(n-1)!}{t^{n-1}},
	\]
	for \(0<s_1<\cdots<s_{n-1}<t\), and is \(0\) otherwise. This is exactly the joint density of the order statistics of \(n-1\) i.i.d. \(U(0,t)\) random variables. Hence
	\[
	(S_1,\dots,S_{n-1}\mid S_n=t)
	\overset{d}{=}
	(Y_{(1)},\dots,Y_{(n-1)}),
	\]
	where \(Y_1,\dots,Y_{n-1}\) are i.i.d. \(U(0,t)\).
\end{proof}

	
	\begin{remark}
		The theorem says that, conditional on knowing exactly how many arrivals occurred by time \(t\), the unordered arrival times behave like independent uniform points on \([0,t]\). After ordering them, we obtain the actual arrival times \(S_1<\cdots<S_n\).
	\end{remark}
	
	
	\begin{theorem}
		Given that \(N(t)=n\), the arrival times \(S_1,\dots,S_n\) have the same distribution as the order statistics of \(n\) independent \(U(0,t)\) random variables.
	\end{theorem}
	
	\begin{proof}
		We want to find the conditional joint density of \(S_1,\dots,S_n\) given \(N(t)=n\). For \(0<s_1<\cdots<s_n<t\), first recall that the joint density of \(S_1,\dots,S_n\) is
		\[
		f_{S_1,\dots,S_n}(s_1,\dots,s_n)
		=
		\lambda^n e^{-\lambda s_n}.
		\]
		This follows from the change of variables \(S_k=T_1+\cdots+T_k\), where \(T_1,\dots,T_n\) are i.i.d. \(\operatorname{Exp}(\lambda)\), with inverse transformation \(T_1=S_1\) and \(T_k=S_k-S_{k-1}\) for \(k=2,\dots,n\). The Jacobian is \(1\), and \(T_1+\cdots+T_n=S_n\).
		
		Now, the event \(N(t)=n\) means that the first \(n\) arrivals occur before time \(t\), but the \((n+1)\)-st arrival occurs after time \(t\). Equivalently, if \(S_n=s_n\), then we need no further arrival in the interval \((s_n,t]\). Thus
		\[
		\mathbb{P}(N(t)=n\mid S_1=s_1,\dots,S_n=s_n)
		=
		\mathbb{P}(T_{n+1}>t-s_n).
		\]
		Since \(T_{n+1}\sim \operatorname{Exp}(\lambda)\), we have
		\[
		\mathbb{P}(T_{n+1}>t-s_n)
		=
		e^{-\lambda(t-s_n)}.
		\]
		Therefore, for \(0<s_1<\cdots<s_n<t\),
		\[
		f_{S_1,\dots,S_n,\,N(t)=n}(s_1,\dots,s_n)
		=
		\lambda^n e^{-\lambda s_n}e^{-\lambda(t-s_n)}
		=
		\lambda^n e^{-\lambda t}.
		\]
		
		Also, since \(N(t)\sim \operatorname{Poisson}(\lambda t)\),
		\[
		\mathbb{P}(N(t)=n)
		=
		e^{-\lambda t}\frac{(\lambda t)^n}{n!}.
		\]
		Hence the conditional joint density of \(S_1,\dots,S_n\) given \(N(t)=n\) is
		\[
		f_{S_1,\dots,S_n\mid N(t)=n}(s_1,\dots,s_n)
		=
		\frac{\lambda^n e^{-\lambda t}}
		{e^{-\lambda t}(\lambda t)^n/n!}
		=
		\frac{n!}{t^n},
		\]
		for \(0<s_1<\cdots<s_n<t\), and is \(0\) otherwise.
		
		But if \(Y_1,\dots,Y_n\) are i.i.d. \(U(0,t)\), then their order statistics satisfy
		\[
		f_{Y_{(1)},\dots,Y_{(n)}}(y_1,\dots,y_n)
		=
		\frac{n!}{t^n},
		\qquad 0<y_1<\cdots<y_n<t.
		\]
		Therefore,
		\[
		(S_1,\dots,S_n\mid N(t)=n)
		\overset{d}{=}
		(Y_{(1)},\dots,Y_{(n)}).
		\]
	\end{proof}
	
	
	\begin{corollary}
		Given that \(N(t)=n\), for any \(0\le a\le b\le t\), the number of arrivals in \((a,b]\) has binomial distribution:
		\[
		N(b)-N(a)\mid N(t)=n
		\sim
		\operatorname{Bin}\left(n,\frac{b-a}{t}\right).
		\]
	\end{corollary}
	
	\begin{proof}
		Conditional on \(N(t)=n\), the \(n\) arrival times are distributed as \(n\) independent uniform points on \([0,t]\), after ordering. For each point, the probability of falling in \((a,b]\) is \((b-a)/t\). Hence the number of points in \((a,b]\) has binomial distribution with parameters \(n\) and \((b-a)/t\).
	\end{proof}
	


	
	\begin{remark}
		There are two closely related conditioning results:
		\begin{itemize}[leftmargin=2em]
			\item Given \(N(t)=n\), all \(n\) arrival times before time \(t\) are distributed like the order statistics of \(n\) i.i.d. \(U(0,t)\) variables.
			\item Given \(S_n=t\), the previous \(n-1\) arrival times are distributed like the order statistics of \(n-1\) i.i.d. \(U(0,t)\) variables.
		\end{itemize}
	\end{remark}

	\section{Nonhomogeneous Poisson Processes}
	% =========================================================
	
	In reality, many arrival processes are not stationary. For example, customers may arrive more frequently during lunch hours than in the early morning. Thus the arrival rate may depend on time. This leads to the notion of a nonhomogeneous Poisson process.
	
	\begin{definition}[Nonhomogeneous Poisson process]
		A counting process \(\{N(t):t\ge 0\}\) is called a nonhomogeneous Poisson process with intensity function \(\lambda(t)\ge 0\) if:
		\begin{enumerate}[label=(\roman*),leftmargin=2em]
			\item \(N(0)=0\).
			\item The process has independent increments.
			\item As \(h\downarrow 0\),
			\[
			\mathbb{P}(N(t+h)-N(t)=1)=\lambda(t)h+o(h).
			\]
			\item As \(h\downarrow 0\),
			\[
			\mathbb{P}(N(t+h)-N(t)\ge 2)=o(h).
			\]
		\end{enumerate}
	\end{definition}
	
	\begin{remark}
		The main difference from a homogeneous Poisson process is that the rate is now time-dependent. If \(\lambda(t)\equiv \lambda\) is constant, then we recover the ordinary Poisson process with rate \(\lambda\).
	\end{remark}
	
	\begin{definition}[Mean value function]
		Let \(\{N(t):t\ge 0\}\) be a nonhomogeneous Poisson process with intensity function \(\lambda(t)\). Define
		\[
		m(t):=\int_0^t \lambda(s)\,ds.
		\]
		The function \(m(t)\) is called the mean value function.
	\end{definition}
	
	\begin{theorem}
		Let \(\{N(t):t\ge 0\}\) be a nonhomogeneous Poisson process with intensity function \(\lambda(t)\), and let
		\[
		m(t)=\int_0^t \lambda(u)\,du.
		\]
		Then for \(0\le s<t\),
		\[
		N(t)-N(s)\sim \operatorname{Poisson}(m(t)-m(s)),
		\]
		where
		\[
		m(t)-m(s)=\int_s^t \lambda(u)\,du.
		\]
		In particular,
		\[
		N(t)\sim \operatorname{Poisson}(m(t)).
		\]
	\end{theorem}
	
	\begin{proof}
		We sketch the idea. Partition \([s,t]\) into many small intervals. On a small interval \([u,u+h]\), the probability of one arrival is approximately \(\lambda(u)h\), and the probability of two or more arrivals is \(o(h)\). Adding these small contributions over the interval \([s,t]\), the total expected number of arrivals is approximately
		\[
		\sum \lambda(u_i)\Delta u_i,
		\]
		which converges to
		\[
		\int_s^t \lambda(u)\,du=m(t)-m(s).
		\]
		Using independent increments and the usual Poisson approximation for rare independent events gives
		\[
		N(t)-N(s)\sim \operatorname{Poisson}(m(t)-m(s)).
		\]
	\end{proof}
	
	\begin{corollary}
		For a nonhomogeneous Poisson process,
		\[
		\mathbb{E}[N(t)]=m(t),
		\qquad
		\operatorname{Var}(N(t))=m(t).
		\]
		More generally,
		\[
		\mathbb{E}[N(t)-N(s)]=m(t)-m(s),
		\qquad
		\operatorname{Var}(N(t)-N(s))=m(t)-m(s).
		\]
	\end{corollary}
	
	\begin{proof}
		Since \(N(t)\sim \operatorname{Poisson}(m(t))\), its mean and variance are both equal to \(m(t)\). Similarly, \(N(t)-N(s)\sim \operatorname{Poisson}(m(t)-m(s))\).
	\end{proof}
	
	\begin{remark}
		A nonhomogeneous Poisson process has independent increments, but generally does not have stationary increments. Indeed, the distribution of \(N(t)-N(s)\) depends on
		\[
		\int_s^t \lambda(u)\,du,
		\]
		not only on the length \(t-s\).
	\end{remark}
	
	\begin{proposition}[Time-dependent thinning]
		Let \(\{N(t):t\ge 0\}\) be a homogeneous Poisson process with rate \(\lambda\). Suppose that an event occurring at time \(s\) is kept with probability \(p(s)\), independently of all other events, where \(0\le p(s)\le 1\). Assume that \(p\) is continuous. Let \(N_p(t)\) be the number of kept events by time \(t\). Then \(\{N_p(t):t\ge 0\}\) is a nonhomogeneous Poisson process with intensity function
		\[
		\lambda_p(t)=\lambda p(t).
		\]
	\end{proposition}
	
	\begin{proof}
		Clearly \(N_p(0)=0\). Since the original process has independent increments and the keeping decisions are made independently for different events, the thinned process also has independent increments.
		
		Fix \(t\ge 0\), and for small \(h>0\) write
		\[
		\Delta N=N(t+h)-N(t),
		\qquad
		\Delta N_p=N_p(t+h)-N_p(t).
		\]
		Thus \(\Delta N\) is the number of original events in \((t,t+h]\), while \(\Delta N_p\) is the number of kept events in \((t,t+h]\).
		
		We first estimate the probability of exactly one kept event. Decomposing according to the number of original events gives
		\[
		\mathbb{P}(\Delta N_p=1)
		=
		\mathbb{P}(\Delta N_p=1,\Delta N=1)
		+
		\mathbb{P}(\Delta N_p=1,\Delta N\ge 2).
		\]
		The second term is negligible, since
		\[
		\mathbb{P}(\Delta N_p=1,\Delta N\ge 2)
		\le
		\mathbb{P}(\Delta N\ge 2)
		=
		o(h).
		\]
		
		Now consider the first term. Conditional on \(\Delta N=1\), the unique event time in \((t,t+h]\) is uniformly distributed on \((t,t+h]\). Hence the conditional probability that this unique event is kept is
		\[
		\frac{1}{h}\int_t^{t+h}p(s)\,ds.
		\]
		Therefore,
		\[
		\mathbb{P}(\Delta N_p=1,\Delta N=1)
		=
		\mathbb{P}(\Delta N=1)\frac{1}{h}\int_t^{t+h}p(s)\,ds.
		\]
		Since \(N\) is a homogeneous Poisson process with rate \(\lambda\), we have \(\mathbb{P}(\Delta N=1)=\lambda h+o(h)\). Also, by continuity of \(p\) at \(t\),
		\[
		\frac{1}{h}\int_t^{t+h}p(s)\,ds=p(t)+o(1).
		\]
		Thus
		\[
		\mathbb{P}(\Delta N_p=1,\Delta N=1)
		=
		(\lambda h+o(h))(p(t)+o(1))
		=
		\lambda p(t)h+o(h).
		\]
		Combining this with the negligible contribution from \(\Delta N\ge 2\), we get
		\[
		\mathbb{P}(\Delta N_p=1)
		=
		\lambda p(t)h+o(h).
		\]
		
		Next, we estimate the probability of two or more kept events. Since kept events can only come from original events, we have
		\[
		\{\Delta N_p\ge 2\}\subseteq \{\Delta N\ge 2\}.
		\]
		Therefore,
		\[
		\mathbb{P}(\Delta N_p\ge 2)
		\le
		\mathbb{P}(\Delta N\ge 2)
		=
		o(h).
		\]
		
		Hence \(N_p\) satisfies
		\[
		\mathbb{P}(N_p(t+h)-N_p(t)=1)=\lambda p(t)h+o(h),
		\]
		and
		\[
		\mathbb{P}(N_p(t+h)-N_p(t)\ge 2)=o(h).
		\]
		Together with \(N_p(0)=0\) and independent increments, this shows that \(N_p\) is a nonhomogeneous Poisson process with intensity \(\lambda_p(t)=\lambda p(t)\).
	\end{proof}
	
	
	\begin{remark}
		This gives one way to generate a nonhomogeneous Poisson process from a homogeneous one. Start with a Poisson process of rate \(\lambda\), and keep an event at time \(t\) with probability \(p(t)\). The resulting process has intensity \(\lambda p(t)\).
	\end{remark}
	
	\begin{proposition}[Dominating process construction]
		Suppose \(\lambda(t)\le \lambda\) for all \(t\ge 0\), where \(\lambda\) is a constant. Let \(\{N^*(t):t\ge 0\}\) be a homogeneous Poisson process with rate \(\lambda\). If each event occurring at time \(t\) is kept with probability
		\[
		p(t)=\frac{\lambda(t)}{\lambda},
		\]
		independently of all other events, then the kept process is a nonhomogeneous Poisson process with intensity \(\lambda(t)\).
	\end{proposition}
	
	\begin{proof}
		This follows directly from the time-dependent thinning result, since the thinned intensity is
		\[
		\lambda p(t)=\lambda\cdot \frac{\lambda(t)}{\lambda}=\lambda(t).
		\]
	\end{proof}
	
	\begin{remark}
		This method is often called thinning or rejection sampling for nonhomogeneous Poisson processes. One first generates a homogeneous Poisson process with a rate large enough to dominate \(\lambda(t)\), and then accepts each event with probability \(\lambda(t)/\lambda\).
	\end{remark}
	
\begin{proposition}[Superposition]
	Let \(\{N_1(t):t\ge 0\}\) and \(\{N_2(t):t\ge 0\}\) be independent
	nonhomogeneous Poisson processes with intensity functions
	\(\lambda_1(t)\) and \(\lambda_2(t)\). Define
	\[
	N(t)=N_1(t)+N_2(t).
	\]
	Then \(\{N(t):t\ge 0\}\) is a nonhomogeneous Poisson process with
	intensity function
	\[
	\lambda(t)=\lambda_1(t)+\lambda_2(t).
	\]
\end{proposition}

\begin{proof}
	The process \(N\) is a counting process and \(N(0)=0\). Since
	\(N_1\) and \(N_2\) have independent increments and are independent
	of each other, \(N\) also has independent increments.
	
	It remains to verify the two small-time conditions. Write
	\(\Delta N_i = N_i(t+h)-N_i(t)\) for \(i=1,2\).
	
	\medskip
	\noindent\textbf{Condition 1: } \(\mathbb{P}(N(t+h)-N(t)=1)
	=\lambda_1(t)h+\lambda_2(t)h+o(h)\).
	
	\medskip
	Exactly one arrival in \(\Delta N = \Delta N_1+\Delta N_2\) occurs
	in one of the following mutually exclusive cases:
	\begin{itemize}
		\item \textbf{Case A:} \(\Delta N_1=1\) and \(\Delta N_2=0\).
		\item \textbf{Case B:} \(\Delta N_1=0\) and \(\Delta N_2=1\).
		\item \textbf{Remaining cases:} at least one process contributes
		\(\ge 2\) arrivals, or both processes each contribute
		\(\ge 1\) arrival. Each such case has probability
		\(\le \mathbb{P}(\Delta N_i\ge 2)=o(h)\) for some \(i\),
		and so contributes only \(o(h)\) in total.
	\end{itemize}
	Therefore,
	\[
	\mathbb{P}(\Delta N=1)
	=
	\mathbb{P}(\Delta N_1=1,\,\Delta N_2=0)
	+
	\mathbb{P}(\Delta N_1=0,\,\Delta N_2=1)
	+o(h).
	\]
	Since \(N_1\) and \(N_2\) are independent, each joint probability
	factors:
	\begin{align*}
		\mathbb{P}(\Delta N_1=1,\,\Delta N_2=0)
		&= \mathbb{P}(\Delta N_1=1)\cdot\mathbb{P}(\Delta N_2=0) \\
		&= \bigl[\lambda_1(t)h+o(h)\bigr]
		\cdot\bigl[1-\lambda_2(t)h+o(h)\bigr] \\
		&= \lambda_1(t)h+o(h),
	\end{align*}
	where the \(h^2\) cross-term is absorbed into \(o(h)\). Similarly,
	\[
	\mathbb{P}(\Delta N_1=0,\,\Delta N_2=1)
	= \lambda_2(t)h+o(h).
	\]
	Summing the two cases gives
	\[
	\mathbb{P}(N(t+h)-N(t)=1)
	= \lambda_1(t)h+\lambda_2(t)h+o(h).
	\]
	
	\medskip
	\noindent\textbf{Condition 2: }
	\(\mathbb{P}(N(t+h)-N(t)\ge 2)=o(h)\).
	
	\medskip
	Two or more arrivals in \(N\) require at least two arrivals across
	\(N_1\) and \(N_2\) combined. Hence
	\[
	\{\Delta N\ge 2\}
	\;\subseteq\;
	\{\Delta N_1\ge 2\}\cup\{\Delta N_2\ge 2\}
	\cup\{\Delta N_1\ge 1,\,\Delta N_2\ge 1\}.
	\]
	Each of the three events on the right has probability \(o(h)\):
	the first two by the small-time condition for \(N_1\) and \(N_2\),
	and the third because
	\[
	\mathbb{P}(\Delta N_1\ge 1,\,\Delta N_2\ge 1)
	\le \mathbb{P}(\Delta N_1\ge 1)\cdot\mathbb{P}(\Delta N_2\ge 1)
	= O(h)\cdot O(h) = o(h).
	\]
	Therefore \(\mathbb{P}(\Delta N\ge 2)=o(h)\).
	
	\medskip
	Together with \(N(0)=0\) and independent increments, the two
	small-time conditions confirm that \(N\) is a nonhomogeneous
	Poisson process with intensity \(\lambda_1(t)+\lambda_2(t)\).
\end{proof}

	
	\begin{remark}
		More generally, the sum of finitely many independent nonhomogeneous Poisson processes is again a nonhomogeneous Poisson process, with intensity equal to the sum of the individual intensities.
	\end{remark}
	
	\begin{proposition}[Source of an event in a superposed process]
		Let \(N_1\) and \(N_2\) be independent nonhomogeneous Poisson processes with intensities \(\lambda_1(t)\) and \(\lambda_2(t)\). Let \(N=N_1+N_2\). Given that an event of \(N\) occurs at time \(t\), the probability that it came from process \(N_1\) is
		\[
		\frac{\lambda_1(t)}{\lambda_1(t)+\lambda_2(t)}.
		\]
		Similarly, the probability that it came from process \(N_2\) is
		\[
		\frac{\lambda_2(t)}{\lambda_1(t)+\lambda_2(t)}.
		\]
	\end{proposition}
	
	\begin{proof}
		On a small interval \((t,t+h]\), the probability of an \(N_1\)-event is approximately \(\lambda_1(t)h\), while the probability of an event from either process is approximately \((\lambda_1(t)+\lambda_2(t))h\). Hence
		\[
		\mathbb{P}(\text{event is from }N_1\mid \text{an event occurs at time }t)
		=
		\frac{\lambda_1(t)}{\lambda_1(t)+\lambda_2(t)}.
		\]
	\end{proof}
	
\begin{proposition}[Time-dependent classification]
	Let \(\{N(t):t\ge 0\}\) be a homogeneous Poisson process with rate
	\(\lambda\). Suppose each event occurring at time \(s\) is classified
	as type \(i\) with probability \(p_i(s)\), where \(i=1,\dots,k\),
	\[
	p_1(s)+\cdots+p_k(s)=1.
	\]
	Let \(N_i(t)\) be the number of type \(i\) events by time \(t\).
	Then \(N_1,\dots,N_k\) are independent nonhomogeneous Poisson
	processes with intensity functions
	\[
	\lambda_i(t)=\lambda p_i(t),
	\qquad i=1,\dots,k.
	\]
\end{proposition}

\begin{proof}
	For a fixed type \(i\), the process \(N_i\) is obtained by keeping
	an event at time \(t\) with probability \(p_i(t)\). By
	time-dependent thinning, \(N_i\) is a nonhomogeneous Poisson
	process with intensity \(\lambda p_i(t)\).
	
	To see independence, condition on the total number of events in
	\((0,t]\). Given \(N(t)=n\), the \(n\) event times are distributed
	as order statistics of \(n\) independent uniform random variables
	on \((0,t]\). In particular, each individual event time \(S\),
	before ordering, is uniformly distributed on \((0,t]\) with density
	\[
	f_S(s) = \frac{1}{t}, \qquad 0 < s < t.
	\]
	Since an event arriving at time \(s\) is classified as type \(i\)
	with probability \(p_i(s)\), the law of total probability gives
	\[
	\mathbb{P}(\text{event is type }i)
	= \int_0^t p_i(s)\,f_S(s)\,ds
	= \frac{1}{t}\int_0^t p_i(s)\,ds.
	\]
	The classification decisions are independent across events. This
	gives a multinomial splitting, and after mixing over \(N(t)\), the
	processes \(N_1,\dots,N_k\) become independent Poisson processes
	with mean functions
	\[
	m_i(t)=\int_0^t \lambda p_i(s)\,ds.
	\]
	Thus \(N_i\) has intensity \(\lambda p_i(t)\).
\end{proof}

	
	\begin{remark}
		The previous proposition generalizes ordinary thinning. If \(p_i(t)\) is constant in time, then each type process is homogeneous Poisson. If \(p_i(t)\) changes with time, then each type process becomes nonhomogeneous.
	\end{remark}
	
	\begin{example}
		Suppose customers arrive at a hotdog stand according to a nonhomogeneous Poisson process with intensity
		\[
		\lambda(t)=5+5t,
		\qquad t\ge 0,
		\]
		where \(t\) is measured in hours after opening. Then the mean value function is
		\[
		m(t)=\int_0^t (5+5s)\,ds=5t+\frac{5}{2}t^2.
		\]
		Hence the number of arrivals between times \(a\) and \(b\), where \(0\le a<b\), has distribution
		\[
		N(b)-N(a)\sim \operatorname{Poisson}(m(b)-m(a)).
		\]
		That is,
		\[
		N(b)-N(a)\sim \operatorname{Poisson}\left(\int_a^b (5+5s)\,ds\right).
		\]
		Therefore, the probability of no arrivals during \((a,b]\) is
		\[
		\mathbb{P}(N(b)-N(a)=0)
		=
		\exp\left\{-\int_a^b (5+5s)\,ds\right\}.
		\]
	\end{example}
	
	\begin{example}
		For the same hotdog stand, the expected number of arrivals between \(8{:}30\) and \(8{:}50\) is
		\[
		\int_{a}^{b}\lambda(s)\,ds,
		\]
		where \(a\) and \(b\) are the corresponding times measured from the chosen origin. The probability of no arrivals during that interval is
		\[
		\exp\left\{-\int_a^b \lambda(s)\,ds\right\}.
		\]
	\end{example}
